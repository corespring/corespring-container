@(namespace : String, loadSession : Call, submitSession:Call)

angular.module('@namespace', []);

angular.module('@namespace').factory('PlayerServices', ['$http', function($http){

var def = {
    submitSession: function(session, onSuccess, onFailure){
      $http['@submitSession.method.toLowerCase']('@submitSession.url', session)
       .success(
         function(data, status, headers, config) {
           onSuccess(data)
         })
       .error(
         function(data, status, headers, config) {
           console.log("error");
           if( onFailure ) {
            onFailure(data);
           }
         }
       );
    },
    loadSession: function(onSuccess, onFailure) {
      $http['@loadSession.method.toLowerCase']('@loadSession.url')
        .success(function(data, status, headers, config){
          console.log("[player-services] load session successful");
          onSuccess(data);
        })
        .error(function(data, status, headers, config){
          if(onFailure){
            onFailure(data);
          }
        })
    }
  }
  return def
}]);
