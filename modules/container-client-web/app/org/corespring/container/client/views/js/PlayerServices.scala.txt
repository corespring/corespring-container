@(namespace : String, loadSession : Call, submitSession:Call, saveSession: Call)

angular.module('@namespace', []);

angular.module('@namespace').factory('PlayerServices', ['$http', function($http){

  var persist = function(method, url){

    return function(session, onSuccess, onFailure){

      $http[method](url, session)
       .success(
         function(data, status, headers, config) {
           onSuccess(data)
         })
       .error(
         function(data, status, headers, config) {
           console.log("error");
           if( onFailure ) {
            onFailure(data);
           }
         }
       );
    };
 };

var def = {
    submitSession: persist('@submitSession.method.toLowerCase', '@submitSession.url'),
    saveSession: persist('@saveSession.method.toLowerCase', '@saveSession.url'),
    loadSession: function(onSuccess, onFailure) {
      $http['@loadSession.method.toLowerCase']('@loadSession.url')
        .success(function(data, status, headers, config){
          console.log("[player-services] load session successful");
          onSuccess(data);
        })
        .error(function(data, status, headers, config){
          if(onFailure){
            onFailure(data);
          }
        })
    }
  }
  return def
}]);
